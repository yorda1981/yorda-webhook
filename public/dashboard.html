async function refreshData() { if (!token) return alert("Error: Token de acceso no válido."); try { // 1. Cargar tasas primero y de forma independiente const resTasas = await fetch(`/admin/tasas?token=${token}`); const tasas = await resTasas.json(); console.log("📊 TASAS RECIBIDAS:", tasas); // Mostrar tasas de inmediato aunque falle el resto Object.keys(tasas).forEach(key => { const input = document.getElementById(key); if (input) { input.value = tasas[key]; } }); // 2. Cargar el resto por separado con bloques try-catch individuales let clientes = []; let stats = { totalOperaciones: 0, volumenTotal: 0, pendientes: 0 }; let operaciones = []; try { const resC = await fetch(`/admin/clientes?token=${token}`); clientes = await resC.json(); } catch (e) { console.error("❌ Error clientes:", e); } try { const resS = await fetch(`/admin/stats?token=${token}`); stats = await resS.json(); } catch (e) { console.error("❌ Error stats:", e); } try { const resO = await fetch(`/admin/operaciones?token=${token}`); operaciones = await resO.json(); } catch (e) { console.error("❌ Error operaciones:", e); } // --- RENDERIZADO DE DATOS (Solo si se obtuvieron correctamente) --- // Cards Superiores const listaClientes = Array.isArray(clientes) ? clientes : Object.values(clientes); document.getElementById("stat-clientes").innerText = listaClientes.length; document.getElementById("stat-ops").innerText = stats.totalOperaciones || 0; document.getElementById("stat-vol").innerText = Number(stats.volumenTotal || 0).toLocaleString('pt-BR', { style: 'currency', currency: 'BRL' }); document.getElementById("stat-pend").innerText = stats.pendientes || 0; // Tabla de Pendientes const tablaPendientes = document.getElementById("tablaPendientes"); const pendientes = operaciones.filter(o => o.status === "pendiente"); tablaPendientes.innerHTML = ""; if (pendientes.length === 0) { tablaPendientes.innerHTML = 'Todo al día. No hay pagos pendientes.'; } else { pendientes.forEach(op => { tablaPendientes.innerHTML += ` ${new Date(op.created_at).toLocaleDateString('pt-BR')}
${new Date(op.created_at).toLocaleTimeString('pt-BR')} ${op.nombre} R$ ${op.monto} ${op.tipo} `; }); } // Ranking de Clientes const tbodyCli = document.getElementById("clientesTable"); tbodyCli.innerHTML = ""; const confirmadas = operaciones.filter(o => o.status === "confirmada"); const ranking = listaClientes.map(c => { const totalConfirmado = confirmadas .filter(o => o.phone === c.phone) .reduce((acc, o) => acc + Number(o.monto), 0); return { ...c, totalReal: totalConfirmado }; }).sort((a, b) => b.totalReal - a.totalReal); ranking.forEach(c => { tbodyCli.innerHTML += ` ${c.nombre || "Desconocido"} ${c.totalReal >= 10000 ? '⭐' : ''} ${c.phone ? c.phone.split('@')[0] : "-"} R$ ${c.totalReal.toLocaleString('pt-BR')} ${c.tipo_favorito || "-"} ${c.totalReal >= 10000 ? '⭐ VIP' : 'Estándar'} `; }); } catch (e) { console.error("Error crítico en refreshData:", e); } }
